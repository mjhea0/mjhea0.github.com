function updateServiceCount(serviceName) {
  return new Promise((resolve, reject) => {
    var params = {
      service: serviceName,
      desiredCount: 0,
      cluster: CLUSTER_NAME
    };
    ecs.updateService(params, (err, data) => {
      if (err) { reject(err); }
      resolve(data);
    });
  });
}

function removeService(serviceName) {
  return new Promise((resolve, reject) => {
    var params = {
      service: serviceName,
      cluster: CLUSTER_NAME
    };
    ecs.deleteService(params, (err, data) => {
      if (err) { reject(err); }
      resolve(data);
    });
  });
}
