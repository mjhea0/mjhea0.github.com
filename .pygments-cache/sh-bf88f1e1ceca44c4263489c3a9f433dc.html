create_target_group() {
	echo "Creating target group..."
  command="aws elbv2 create-target-group --name "$TARGET_GROUP-$1" --protocol HTTP --port $2 --vpc-id $VPC_ID --health-check-path $3"
	if [[ $( | $JQ ".TargetGroups[0].TargetGroupName") == "$TARGET_GROUP-$1" ]]; then
		echo "Target group created!"
	else
		echo "Error creating target group."
		return 1
  fi
}
