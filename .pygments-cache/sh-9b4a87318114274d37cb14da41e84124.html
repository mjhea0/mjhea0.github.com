create_service_and_deploy() {
	echo "Creating service..."
  if [[ $(aws ecs create-service --cluster $ECS_CLUSTER --service-name $ECS_SERVICE --task-definition $revision --desired-count 1 | $JQ ".service.taskDefinition") == $revision ]]; then
		echo "Service created!"
	else
		echo "Error creating service service."
		return 1
  fi
}
