create_listener() {
	echo "Creating listener..."
	if [[ $(aws elbv2 create-listener --load-balancer-arn $LOAD_BALANCER_ARN --protocol HTTP --port 30334 --default-actions Type=forward,TargetGroupArn=$SAMPLE_TARGET_GROUP_ARN | $JQ ".Listeners[0].Port") == $port ]]; then
		echo "Listener created!"
	else

  fi
}
