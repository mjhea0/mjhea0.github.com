function addListener(port) {
  return new Promise((resolve, reject) => {
    var params = {
      DefaultActions: [
        {
          TargetGroupArn: DEFAULT_TARGET_GROUP_ARN,
          Type: "forward"
        }
      ],
      LoadBalancerArn: LOAD_BALANCER_ARN,
      Port: port,
      Protocol: "HTTP"
    };
    elbv2.createListener(params, (err, data) => {
      if (err) { reject(err); }
      resolve(data);
    });
  });
}
