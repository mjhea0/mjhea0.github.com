tag_and_push_images() {
	echo "Tagging and pushing images..."
	# tag
	docker tag ${ECR_BASE}web ${ECR_URI}/${ECR_BASE}web:test
	docker tag ${ECR_BASE}movies-api ${ECR_URI}/${ECR_BASE}movies-api:test
	docker tag ${ECR_BASE}movies-db ${ECR_URI}/${ECR_BASE}movies-db:test
	docker tag ${ECR_BASE}swagger ${ECR_URI}/${ECR_BASE}swagger:test
	docker tag ${ECR_BASE}users-api ${ECR_URI}/${ECR_BASE}users-api:test
	docker tag ${ECR_BASE}users-db ${ECR_URI}/${ECR_BASE}users-db:test
	# push
	docker push ${ECR_URI}/microservice-movies-web:test
	docker push ${ECR_URI}/${ECR_BASE}web:test
	docker push ${ECR_URI}/${ECR_BASE}movies-api:test
	docker push ${ECR_URI}/${ECR_BASE}movies-db:test
	docker push ${ECR_URI}/${ECR_BASE}swagger:test
	docker push ${ECR_URI}/${ECR_BASE}users-api:test
	docker push ${ECR_URI}/${ECR_BASE}users-db:test
	echo "Images tagged and pushed!"
}
