function addRule(targetGroup, pattern, listener, priority) {
  return new Promise((resolve, reject) => {
    var params = {
      Actions: [
        {
          TargetGroupArn: targetGroup,
          Type: 'forward'
        }
     ],
     Conditions: [
      {
        Field: 'path-pattern',
        Values: [pattern]
      }
     ],
     ListenerArn: listener,
     Priority: priority
    };
    elbv2.createRule(params, (err, data) => {
      if (err) { reject(err); }
      resolve(data);
    });
  });
}
