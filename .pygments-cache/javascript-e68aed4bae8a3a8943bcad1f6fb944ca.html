authService.$inject = ['$http'];

function authService($http) {
  /*jshint validthis: true */
  const baseURL = 'http://localhost:3000/auth/';
  this.login = function(user) {
    return $http({
      method: 'POST',
      url: baseURL + 'login',
      data: user,
      headers: {'Content-Type': 'application/json'}
    });
  };
}
