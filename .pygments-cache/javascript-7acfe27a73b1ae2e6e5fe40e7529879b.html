function authStatusController(authService) {
  /*jshint validthis: true */
  const vm = this;
  vm.isLoggedIn = false;
  const token = localStorage.getItem('token');
  if (token) {
    authService.ensureAuthenticated(token)
    .then((user) => {
      if (user.data.status === 'success');
      vm.isLoggedIn = true;
    })
    .catch((err) => {
      console.log(err);
    });      
  }
}
