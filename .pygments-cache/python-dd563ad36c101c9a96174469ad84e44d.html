import sqlite3

sql_command_create_table = """
DROP TABLE IF EXISTS songs;
CREATE TABLE songs(id INTEGER PRIMARY KEY AUTOINCREMENT,
artist TEXT NOT NULL,
title TEXT NOT NULL,
rating INTEGER NOT NULL);
"""


def init_db():
    with sqlite3.connect('songs.db') as connection:
        cursor = connection.cursor()
        cursor.execute(sql_command_create_table)
        connection.commit()
        connection.close()


init_db()
